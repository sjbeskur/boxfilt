#Get the stars psf
psf_size = 30
detected_star = 0
num_psf_star = 0
connectivity = 4
average_psf = np.zeros((psf_size,psf_size))

img_color = cv.cvtColor(img_gray, cv.COLOR_GRAY2BGR) #create a color image so we can draw the colored contour on it
ret, img_binary = cv.threshold(img_gray, 50, 255, cv.THRESH_BINARY)

(numLabels, labels, stats, centroids) = cv.connectedComponentsWithStats(img_binary, connectivity, cv.CV_32S)
for i in range(1, numLabels):
    componentMask = (labels == i).astype("uint8") * 255
    M = cv.moments(componentMask, False);
    area = M['m00'] #moment 00 is the area of the object
    if area > 0:
        detected_star +=1
        contours, hierarchy = cv.findContours(componentMask, cv.RETR_TREE, cv.CHAIN_APPROX_SIMPLE)
        if(cv.contourArea(contours[0])>2):
            cv.drawContours(img_color, contours, -1, (0,255,0), 1)
            cx = int(M['m10']/area)
            cy = int(M['m01']/area)
            psf = img_gray[cy - psf_size//2:cy + psf_size//2, cx - psf_size//2:cx + psf_size//2 ]
            average_psf = average_psf + psf #average out the psf
            num_psf_star += 1

print("number of stars detected = %d" % detected_star)
print("number of psf stars chosen = %d" % num_psf_star)
average_psf = average_psf / num_psf_star / 3    #the bigger the value the less impact the psf has

plt.figure(figsize=(10, 10))
plt.imshow(img_color)

plt.figure(figsize=(10, 10))
plt.imshow(average_psf)
plt.show()
